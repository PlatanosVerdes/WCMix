//Constantes
#define capacidad 3
#define Mujeres 4
#define Hombres 4
#define TRUE 1

//Variables Globales

typedef struct
{
    int value;
    /* Lista de procesos */
} SEMAFORO;

SEMAFORO SemLavabo; //Controla el lavabo (Sección crítica)
SEMAFORO SemHombres; //Controla el acceso de los hombres
SEMAFORO SemMujeres; //Controla el acceso de las mujeres
int NumHombres=0;
int NumMujeres=0;

main()
{
    extern SEMAFORO SemLavabo,SemHombres,SemMujeres;
    int i;
    void hombre(),mujer();

    InitSemaphore(SemLavabo, capacidad);
    InitSemaphore(SemHombres, 1);
    InitSemaphore(SemMujeres, 1);

    for(i=0;i<Mujeres;i++)
    {
        if(!fork())
        {    
            mujer();
        }
    }

    for(i=0;i<Hombres;i++)
    {
        if(!fork())
        {    
            hombre();
        }
    }
}

void hombre()
{
    extern int NumHombres,NumMujeres;
    extern SEMAFORO SemHombres,SemLavabo,SemMujeres;

    while(TRUE)
    {
        wait(SemHombres); // Bloqueo al hombre para comprobar el baño
        if(NumMujeres==0 && NumHombres<capacidad)//Compruebo que puedo entrar
        {
            NumHombres++;//Incremento el número de hombres   
            signal(SemHombres); //Desbloqueo al hombre
            wait(SemLavabo); //Aseguro el baño (Sección crítica)
            /*Seccion critica*/
            signal(SemLavabo); //Desbloqueo el baño
            NumHombres--; //Decremento el número de hombres
        }
        else
        {
            signal(SemMujeres); //Desbloqueo a las mujeres para que entren
        }
    }
}

void mujer()
{
    extern int NumHombres,NumMujeres;
    extern SEMAFORO SemHombres,SemLavabo,SemMujeres;

    while(TRUE)
    {
        wait(SemMujeres);// Bloqueo a la mujer para comprobar el baño
        if(NumHombres==0 && NumMujeres<capacidad)//Compruebo que puedo entrar
        {
            NumMujeres++; //Incremento el número de mujeres
            signal(SemMujeres);//Desbloqueo a la mujer
            wait(SemLavabo); //Aseguro el baño (Sección crítica)
            /*Seccion critica*/
            signal(SemLavabo);//Desbloqueo el baño
            NumMujeres--;//Decremento el número de mujeres
        }
        else
        {
            signal(SemHombres);//Desbloqueo a los hombres para que entren
        }
    }
}
